' ==================== ENTITIES ====================
package "Entities" #LightBlue {
    class User {
        -id: Long
        -username: String
        -passwordHash: String
        -email: String
        -createdAt: LocalDateTime
    }

    class Media {
        -id: Long
        -title: String
        -description: String
        -mediaType: String
        -releaseYear: Integer
        -genres: List<String>
        -ageRestriction: String
        -creatorId: Long
        -createdAt: LocalDateTime
    }

    class Rating {
        -id: Long
        -userId: Long
        -mediaId: Long
        -score: Integer
        -comment: String
        -isConfirmed: Boolean
        -likeCount: Integer
        -createdAt: LocalDateTime
        -updatedAt: LocalDateTime
    }

    class Favorite {
        -id: Long
        -userId: Long
        -mediaId: Long
        -createdAt: LocalDateTime
    }

    class RatingLike {
        -id: Long
        -userId: Long
        -ratingId: Long
        -createdAt: LocalDateTime
    }
}

' ==================== DATA ACCESS ====================
package "Data Access (DAO)" #LightGreen {
    class UserDAO {
        +save(conn, user): void
        +findByUsername(conn, username): User
        +findById(conn, id): User
        +update(conn, user): void
        +findLeaderboard(conn, limit): List
    }

    class MediaDAO {
        +save(conn, media): void
        +findById(conn, id): Media
        +findAll(conn): List<Media>
        +update(conn, media): void
        +delete(conn, id): void
        +findWithFilters(conn, ...): List<Media>
    }

    class RatingDAO {
        +save(conn, rating): void
        +findById(conn, id): Rating
        +findByMediaId(conn, mediaId): List<Rating>
        +findByUserId(conn, userId): List<Rating>
        +update(conn, rating): void
        +delete(conn, id): void
        +existsByUserAndMedia(conn, ...): boolean
    }

    class FavoriteDAO {
        +save(conn, favorite): void
        +deleteByUserAndMedia(conn, ...): void
        +findByUserId(conn, userId): List<Favorite>
        +existsByUserAndMedia(conn, ...): boolean
    }

    class RatingLikeDAO {
        +save(conn, like): void
        +existsByUserAndRating(conn, ...): boolean
        +deleteByUserAndRating(conn, ...): void
        +countByRatingId(conn, ratingId): int
    }
}

' ==================== SERVICES ====================
package "Business Logic (Services)" #LightYellow {
    class UserService {
        -userDAO: UserDAO
        -ratingDAO: RatingDAO
        -mediaDAO: MediaDAO
        +registerUser(username, password, email): User
        +loginUser(username, password): User
        +getUserByUsername(username): User
        +getUserStatistics(userId): UserStatisticsResponse
        +getUserRatings(userId): List<Rating>
        +getLeaderboard(limit): List
        +updateUserProfile(username, email): User
    }

    class AuthService {
        -tokenStore: Map<String, String>
        +generateToken(username): String
        +validateToken(token): boolean
        +getUsernameFromToken(token): String
        +invalidateToken(token): void
    }

    class MediaService {
        -mediaDAO: MediaDAO
        +createMedia(...): Media
        +getAllMedia(): List<Media>
        +searchMedia(...): List<Media>
        +getMediaById(id): Media
        +updateMedia(...): Media
        +deleteMedia(id, userId): void
    }

    class RatingService {
        -ratingDAO: RatingDAO
        -ratingLikeDAO: RatingLikeDAO
        +createRating(userId, mediaId, score, comment): Rating
        +updateRating(ratingId, userId, score, comment): Rating
        +deleteRating(ratingId, userId): void
        +confirmComment(ratingId, userId): Rating
        +calculateAverageRating(mediaId): Double
        +getRatingsByMediaId(mediaId): List<Rating>
        +getRatingsByMediaIdPublic(mediaId, userId): List<Rating>
        +likeRating(ratingId, userId): Rating
        +unlikeRating(ratingId, userId): Rating
    }

    class FavoriteService {
        -favoriteDAO: FavoriteDAO
        -mediaDAO: MediaDAO
        +addFavorite(userId, mediaId): Favorite
        +removeFavorite(userId, mediaId): void
        +getFavoritesByUserId(userId): List<Favorite>
        +isFavorite(userId, mediaId): boolean
    }

    class RecommendationService {
        -ratingDAO: RatingDAO
        -mediaDAO: MediaDAO
        +getRecommendationsByGenre(userId, limit): List<Media>
        +getRecommendationsByContent(userId, limit): List<Media>
    }
}

' ==================== CONTROLLERS ====================
package "HTTP Controllers" #LightCoral {
    class AuthController {
        -authService: AuthService
        -userService: UserService
        +handleRegister(exchange): void
        +handleLogin(exchange): void
    }

    class UserController {
        -userService: UserService
        -authService: AuthService
        -recommendationService: RecommendationService

        +handleLeaderboard(exchange): void
    }

    class MediaController {
        -mediaService: MediaService
        -ratingService: RatingService

        +handleMedia(exchange): void
    }

    class RatingController {
        -ratingService: RatingService
        -authService: AuthService
        -userService: UserService
        +handleRating(exchange): void
    }

    class FavoriteController {
        -favoriteService: FavoriteService
        -authService: AuthService
        -userService: UserService
        +handleFavorite(exchange): void
    }
}
